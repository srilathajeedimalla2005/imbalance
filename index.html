# -------------------------------
# CODING AND DATA ANALYSIS TASK
# -------------------------------

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

from sklearn.datasets import make_classification
from sklearn.model_selection import train_test_split, cross_val_score
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LogisticRegression
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import (
    confusion_matrix, classification_report,
    roc_curve, auc, precision_recall_curve, f1_score
)

from imblearn.over_sampling import SMOTE

# -------------------------------
# 1. DATA PREPARATION
# -------------------------------
# Create imbalanced dataset
X, y = make_classification(n_samples=2000, n_features=20, 
                           n_informative=10, n_redundant=5,
                           n_classes=2, weights=[0.85, 0.15],
                           random_state=42)

print("Class distribution before balancing:", np.bincount(y))

# Split data
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.3, random_state=42, stratify=y)

# Feature scaling
scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test = scaler.transform(X_test)

# -------------------------------
# 2. BALANCING THE DATA
# -------------------------------
sm = SMOTE(random_state=42)
X_train_res, y_train_res = sm.fit_resample(X_train, y_train)

print("Class distribution after SMOTE:", np.bincount(y_train_res))

# -------------------------------
# 3. BASELINE MODELS
# -------------------------------
# Logistic Regression
lr = LogisticRegression(max_iter=500, random_state=42)
lr.fit(X_train_res, y_train_res)

# Random Forest
rf = RandomForestClassifier(n_estimators=100, random_state=42)
rf.fit(X_train_res, y_train_res)

# -------------------------------
# 4. PROPOSED MODEL (DAM-CLF Example)
# -------------------------------
# For demo, simulate DAM-CLF by combining RF + LR predictions
class DAM_CLF:
    def __init__(self, models):
        self.models = models
    
    def fit(self, X, y):
        for model in self.models:
            model.fit(X, y)
    
    def predict_proba(self, X):
        probs = [m.predict_proba(X) for m in self.models]
        avg_probs = np.mean(probs, axis=0)  # averaging
        return avg_probs
    
    def predict(self, X, threshold=0.5):
        probs = self.predict_proba(X)[:,1]
        return (probs >= threshold).astype(int)

dam = DAM_CLF([lr, rf])
dam.fit(X_train_res, y_train_res)

# -------------------------------
# 5. EVALUATION FUNCTION
# -------------------------------
def evaluate_model(model, X_test, y_test, name, threshold=0.5):
    if hasattr(model, "predict_proba"):
        probs = model.predict_proba(X_test)[:,1]
    else:
        probs = model.predict_proba(X_test)[:,1]
        
    preds = (probs >= threshold).astype(int)
    
    print(f"\n{name} Classification Report (threshold={threshold}):")
    print(classification_report(y_test, preds))
    
    # Confusion Matrix
    cm = confusion_matrix(y_test, preds)
    sns.heatmap(cm, annot=True, fmt="d", cmap="Blues")
    plt.title(f"{name} - Confusion Matrix")
    plt.xlabel("Predicted")
    plt.ylabel("Actual")
    plt.show()
    
    # ROC Curve
    fpr, tpr, _ = roc_curve(y_test, probs)
    roc_auc = auc(fpr, tpr)
    plt.plot(fpr, tpr, label=f"{name} AUC={roc_auc:.2f}")
    plt.plot([0,1], [0,1], "k--")
    plt.xlabel("False Positive Rate")
    plt.ylabel("True Positive Rate")
    plt.title(f"{name} - ROC Curve")
    plt.legend()
    plt.show()
    
    # Precision-Recall Curve
    precision, recall, thresholds = precision_recall_curve(y_test, probs)
    plt.plot(recall, precision, label=f"{name}")
    plt.xlabel("Recall")
    plt.ylabel("Precision")
    plt.title(f"{name} - Precision Recall Curve")
    plt.legend()
    plt.show()
    
    return f1_score(y_test, preds)

# -------------------------------
# 6. RUN EVALUATION
# -------------------------------
f1_lr = evaluate_model(lr, X_test, y_test, "Logistic Regression")
f1_rf = evaluate_model(rf, X_test, y_test, "Random Forest")
f1_dam = evaluate_model(dam, X_test, y_test, "Proposed DAM-CLF")

print("\nF1 Scores Comparison:")
print(f"Logistic Regression: {f1_lr:.4f}")
print(f"Random Forest: {f1_rf:.4f}")
print(f"DAM-CLF (Proposed): {f1_dam:.4f}")
